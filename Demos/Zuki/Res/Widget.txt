[Basic]
Desc=Widget

[DimAligned]
Phases=3
Name = Aligned Dim.

1=Set second extension line origin
2=Set dimension location

[Dim3PAngle]
Phases=4
Name=3-P Angle Dim.

1=Set first angle endpoint
2=Set second angle endpoint

[DimRad]
Phases=2
Name=Radius Dim.
0=Select circle or arc
1=Set dimension location [F5=Toggle]

[DimDia]
Phases=2
Name=Diameter Dim.
0=Select circle
1=Set dimension location [F5=Toggle]

[DimAngle]
Phases=3
Name=Angle Dim.
0=Select first line
1=Select second line
2=Set dimension arc location
